Don't translate <rect> when fill is gradient with gradientUnits="userSpaceOnUse".

===

<svg xmlns="http://www.w3.org/2000/svg" viewBox="0 0 80 80">
  <defs>
    <linearGradient id="a" gradientUnits="userSpaceOnUse" x1="10" x2="60" y1="10" y2="10">
      <stop stop-color="blue"/>
      <stop offset="1" stop-color="red"/>
    </linearGradient>
  </defs>
  <rect x="10" y="10" width="50" height="50" fill="url(#a)" transform="translate(4,5)"/>

</svg>
