===

  <rect x="0" y="10" width="5" height="8" fill="red" style="transform:scale(.5)scale(2)"/>
  <rect x="0" y="10" width="5" height="8" fill="red" style="transform:translate(22.3px)translate(-22.3px)"/>
</svg>
