Don't collapse groups with clip-path style property if there is an individual transform property on child.

===

<svg xmlns="http://www.w3.org/2000/svg" viewBox="0 0 100 100.0">
  <clipPath id="a">
    <path d="M20 20L20 80L80 70L60 40z" fill="none" stroke="black"/>
  </clipPath>
  <g style="clip-path:url(#a)">
    <circle cx="50" cy="50" r="30" style="fill:red;translate:2px 3px"/>
  </g>
</svg>

@@@
