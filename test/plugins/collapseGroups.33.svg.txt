Don't collapse groups with clip-path whose child is <use>.

===

<svg xmlns="http://www.w3.org/2000/svg" viewBox="0 0 1000 1000">
  <defs>
    <clipPath id="a" clipPathUnits="userSpaceOnUse">
      <circle cx="25" cy="300" r="150"/>
    </clipPath>
    <symbol id="b" viewBox="0 0 450 300">
      <rect width="450" height="300"/>
    </symbol>
  </defs>
  <g clip-path="url(#a)">
    <use x="100" y="350" width="225" height="150" href="#b"/>
  </g>

</svg>

@@@
