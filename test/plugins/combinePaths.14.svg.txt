Don't merge paths with attribute selector on "d".

===

<svg xmlns="http://www.w3.org/2000/svg" viewBox="0 0 20 20">
  <style>
    .stroke {stroke:blue}
    .stroke[d=&quot;M 1 5 h 10&quot;] {stroke:red}
  </style>
  <g>
    <path d="M 1 5 h 10" class="stroke"/>
    <path d="M 1 7 h 10" class="stroke"/>
  </g>
</svg>

@@@
