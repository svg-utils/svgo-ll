Don't merge paths with pseudo-classes.

===

<svg xmlns="http://www.w3.org/2000/svg" viewBox="0 0 20 20">
  <style>
  path:first-child {stroke:yellow}
  path {stroke:red}
  path.a {stroke:green}
  </style>
  <g>
    <path d="M1 3h10"/>
    <path d="M1 5h10"/>
    <path d="M1 7h10" class="a"/>
    <path d="M1 9h10" class="a"/>
  </g>
</svg>
