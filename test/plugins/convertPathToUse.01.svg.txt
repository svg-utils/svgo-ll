Convert common paths to <use>.

===

<svg xmlns="http://www.w3.org/2000/svg" viewBox="0 0 100 100">
  <path d="M10 10l10 1l10 -1l10 1l10 -1" fill="none" stroke="red"/>
  <path d="M10 10l10 1l10 -1l10 1l10 -1" fill="none" stroke="blue" transform="translate(0 10)"/>
  <path d="M10 10l10 1l10 -1l10 1l10 -1" fill="none" stroke="green" transform="translate(0 20)"/>
</svg>

@@@

<svg xmlns="http://www.w3.org/2000/svg" viewBox="0 0 100 100">
  <use fill="none" stroke="red" href="#a"/>
  <use fill="none" stroke="blue" transform="translate(0 10)" href="#a"/>
  <use fill="none" stroke="green" transform="translate(0 20)" href="#a"/>
  <defs>
    <path id="a" d="M10 10l10 1l10 -1l10 1l10 -1"/>
  </defs>
</svg>
