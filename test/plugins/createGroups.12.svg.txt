Create group with opacity.

===

<svg xmlns="http://www.w3.org/2000/svg" viewBox="0 0 100 100">
  <path d="M10 10h60" stroke="red" opacity=".2"/>
  <path d="M10 20h60" stroke="green" opacity=".2"/>
  <path d="M10 30h60" stroke="blue" opacity=".2"/>
</svg>

<svg xmlns="http://www.w3.org/2000/svg" viewBox="0 0 100 100">
  <g style="opacity:.2">
    <path d="M10 10h60" stroke="red"/>
    <path d="M10 20h60" stroke="green"/>
    <path d="M10 30h60" stroke="blue"/>
  </g>
</svg>
