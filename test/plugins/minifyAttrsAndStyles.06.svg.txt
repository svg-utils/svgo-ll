Don't convert style to attribute if <style> is present.

===

<svg xmlns="http://www.w3.org/2000/svg" viewBox="0 0 30 30">
  <style>.green{fill:green}</style>
  <rect class="green" y="3" width="3" height="2" style="fill:red"/>
</svg>

@@@

<svg xmlns="http://www.w3.org/2000/svg" viewBox="0 0 30 30">
  <style>.green{fill:green}</style>
  <rect class="green" y="3" width="3" height="2" style="fill:red"/>
</svg>
