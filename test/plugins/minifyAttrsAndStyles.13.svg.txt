Shorten transforms even if <style> is present.

===

<svg viewBox="0 0 40 40"
  xmlns="http://www.w3.org/2000/svg">
  <style>rect{fill:blue}</style>
  <linearGradient gradientTransform="translate(2 3)scale(4 5)"/>
  <rect width="1" height="1" style="transform:translate(2px,3px)scale(4,5)"/>
</svg>

@@@

<svg xmlns="http://www.w3.org/2000/svg" viewBox="0 0 40 40">
  <style>rect{fill:blue}</style>
  <linearGradient gradientTransform="matrix(4 0 0 5 2 3)"/>
  <rect width="1" height="1" style="transform:matrix(4,0,0,5,2,3)"/>
</svg>
