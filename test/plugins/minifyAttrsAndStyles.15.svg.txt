Don't convert individual transform properties to attributes.

===

<svg xmlns="http://www.w3.org/2000/svg" viewBox="0 0 15 15.0">
  <g style="translate:1px 2px">
    <rect width="2" height="1.5" transform="rotate(10)"/>
  </g>
  
</svg>

@@@

<svg xmlns="http://www.w3.org/2000/svg" viewBox="0 0 15 15">
  <g style="translate:1px 2px">
    <rect width="2" height="1.5" transform="rotate(10)"/>
  </g>
</svg>
