Minify class names correctly when <style> contains unreferenced class name.

===

<svg xmlns="http://www.w3.org/2000/svg" viewBox="0 0 32 32">
  <style>.a{fill:red}.c{fill:green}.b{fill:blue}</style>
  <rect class="a" y="1" width="2" height="2"/>
  <rect class="c" y="4" width="2" height="2"/>
  <rect class="c" y="7" width="2" height="2"/>
</svg>

@@@

<svg xmlns="http://www.w3.org/2000/svg" viewBox="0 0 32 32">
  <style>.b{fill:red}.a{fill:green}.c{fill:blue}</style>
  <rect class="b" y="1" width="2" height="2"/>
  <rect class="a" y="4" width="2" height="2"/>
  <rect class="a" y="7" width="2" height="2"/>
</svg>
