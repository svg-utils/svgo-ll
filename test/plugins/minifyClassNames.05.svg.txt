===

  <style>.b{fill:red}.a{opacity:.1}</style>
  <rect class="b" y="1" width="2" height="2"/>
  <rect class="b a" y="4" width="2" height="2"/>
  <rect class="a b" y="7" width="2" height="2"/>
</svg>
