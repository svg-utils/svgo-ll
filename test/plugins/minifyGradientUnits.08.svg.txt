If a gradient is only referenced as a template, move all common attributes to template.

===

<svg xmlns="http://www.w3.org/2000/svg" viewBox="0 0 150 150">
  <defs>
    <linearGradient id="linearGradient5" href="#linearGradient1" x1="25" y1="30" x2="80" y2="30" spreadMethod="reflect"/>
    <linearGradient id="linearGradient6" href="#linearGradient1" x1="25" y1="30" x2="80" y2="30" spreadMethod="reflect"/>
    <linearGradient id="linearGradient1" x1="40" y1="30" x2="80" y2="70" gradientUnits="userSpaceOnUse">
      <stop style="stop-color:#f57900"/>
      <stop style="stop-color:#8ae234" offset="1"/>
    </linearGradient>
  </defs>
  <rect style="fill:url(#linearGradient5)" width="55" height="40" x="45" y="30"/>

</svg>
