Don't remove <style> from <defs>.

===

<svg xmlns="http://www.w3.org/2000/svg" viewBox="0 0 20 20">
  <defs>
    <style>
          path{stroke-width:.5}
    </style>
    <path id="a" style="fill:none;stroke:black;" d="M2 2H10"/>
    <path id="b" style="fill:none;stroke:black;" d="M2 4H10"/>
  </defs>
  <use href="#b"/>
</svg>

@@@
