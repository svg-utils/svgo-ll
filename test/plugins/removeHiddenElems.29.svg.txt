===

  <rect fill="url(#a)" width="2" height="2"/>
  <rect fill="url(#b)" width="2" height="2" y="5"/>
  <defs>
    <g>
      <g>
        <linearGradient id="a">
          <stop offset="5%" stop-color="gold" />
        </linearGradient>
      </g>
    </g>
    <linearGradient id="b">
      <stop offset="5%" stop-color="red" />
    </linearGradient>
  </defs>
</svg>
