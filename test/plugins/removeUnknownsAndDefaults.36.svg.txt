Account for styles when removing attributes from <use>.

===

<svg viewBox="0 0 40 10"
  xmlns="http://www.w3.org/2000/svg">
  <circle id="c1" cx="5" cy="5" r="4" fill="red"/>
  <use href="#c1" x="10" style="fill:blue"/>
  <use href="#c1" x="20" style="fill-opacity:.3"/>
</svg>

@@@

  <use href="#c1" x="20" style="fill-opacity:.3"/>
