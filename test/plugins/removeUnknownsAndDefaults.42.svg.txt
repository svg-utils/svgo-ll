clip-path attribute can be removed when it is identical to parent.

===

  <clipPath id="a">
    <path d="M20 20L20 80L80 70L60 40z" fill="none" stroke="black"/>
  </clipPath>
  <g clip-path="url(#a)" transform="translate(15)">
    <g clip-path="url(#a)" transform="translate(15)">
      <circle cx="50" cy="50" r="30" fill="red" clip-path="url(#a)"/>
    </g>
  </g>
</svg>
