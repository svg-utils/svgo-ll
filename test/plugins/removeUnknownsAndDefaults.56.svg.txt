Preserve <use>d colors when <use>s are nested.

===

<svg xmlns="http://www.w3.org/2000/svg" viewBox="0 0 80 80">
  <defs>
    <g id="a">
      <rect width="18" height="18" fill="currentColor"/>
    </g>
    <use id="b" href="#a"/>
    <g id="c">
      <rect width="18" height="18" fill="currentColor"/>
    </g>
    <use id="d" href="#c" color="yellow"/>
  </defs>
  <use href="#b" x="2" y="2" color="red"/>
  <use href="#d" x="22" y="2" color="red"/>
</svg>
