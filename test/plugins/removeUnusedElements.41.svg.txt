Convert unreferenced <symbol> to defs.

===

<svg viewBox="-10 -10 70 70"
  xmlns="http://www.w3.org/2000/svg">
  <symbol id="a">
    <rect id="b" width="50" height="50"/>
  </symbol>
  <use href="#b"/>
</svg>

@@@

<svg viewBox="-10 -10 70 70" xmlns="http://www.w3.org/2000/svg">
  <defs>
    <rect id="b" width="50" height="50"/>
  </defs>
  <use href="#b"/>
</svg>
