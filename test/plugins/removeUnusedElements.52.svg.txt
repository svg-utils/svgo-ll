Promote nested <defs>, even if it has an id.

===

<svg viewBox="0 0 480 360"
  xmlns="http://www.w3.org/2000/svg">
  <defs>
    <mask maskUnits="userSpaceOnUse" id="mask5405-3">
      <rect id="b" style="fill:url(#a)" width="100" height="100"/>
    </mask>
    <defs id="whatever">
      <linearGradient id="a">
        <stop offset="5%" stop-color="gold" />
      </linearGradient>
    </defs>
  </defs>
  <use href="#b"/>
</svg>
