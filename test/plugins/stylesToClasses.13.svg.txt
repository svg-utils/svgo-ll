Don't convert to classes for non-rendering elements.

===

<svg viewBox="0 0 40.0 40"
  xmlns="http://www.w3.org/2000/svg">
  <linearGradient gradientTransform="translate(2 3)scale(4 5)"/>
  <linearGradient gradientTransform="translate(2 3)scale(4 5)"/>
  <linearGradient gradientTransform="translate(2 3)scale(4 5)"/>
</svg>

<svg viewBox="0 0 40 40" xmlns="http://www.w3.org/2000/svg">

  <linearGradient gradientTransform="translate(2 3)scale(4 5)"/>
  <linearGradient gradientTransform="translate(2 3)scale(4 5)"/>
</svg>
